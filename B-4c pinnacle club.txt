#include <iostream>
using namespace std;

struct Node {
    int prn;
    string name;
    Node* next;
};

class PinnacleClub {
private:
    Node* head;

public:
    PinnacleClub() {
        head = nullptr;
    }

    void addPresident(int prn, string name) {
        Node* newNode = new Node{prn, name, head};
        head = newNode;
        cout << "President added successfully.\n";
    }

    void addSecretary(int prn, string name) {
        Node* newNode = new Node{prn, name, nullptr};
        if (head == nullptr) {
            head = newNode;
            cout << "Secretary added successfully as first node.\n";
            return;
        }
        Node* temp = head;
        while (temp->next != nullptr)
            temp = temp->next;
        temp->next = newNode;
        cout << "Secretary added successfully.\n";
    }

    void addMember(int prn, string name) {
        if (head == nullptr || head->next == nullptr) {
            cout << "Please add President and Secretary first.\n";
            return;
        }
        Node* newNode = new Node{prn, name, nullptr};
        Node* temp = head;
        while (temp->next->next != nullptr)
            temp = temp->next;

        newNode->next = temp->next;
        temp->next = newNode;
        cout << "Member added successfully.\n";
    }

    void displayMembers() {
        if (head == nullptr) {
            cout << "No members in this club.\n";
            return;
        }

        cout << "\n--- Pinnacle Club Members ---\n";
        Node* temp = head;
        while (temp != nullptr) {
            cout << temp->prn << " - " << temp->name;
            if (temp == head)
                cout << " (President)";
            else if (temp->next == nullptr)
                cout << " (Secretary)";
            cout << endl;
            temp = temp->next;
        }
    }

    void countMembers() {
        int count = 0;
        Node* temp = head;
        while (temp != nullptr) {
            count++;
            temp = temp->next;
        }
        cout << "Total number of members in this club: " << count << endl;
    }

    void concatenate(PinnacleClub& other) {
        if (head == nullptr) {
            head = other.head;
            return;
        }

        Node* temp = head;
        while (temp->next != nullptr)
            temp = temp->next;

        temp->next = other.head;
        cout << "Both divisions concatenated successfully.\n";
    }
};


int main() {
    PinnacleClub divA, divB; 
    int choice, prn;
    string name;

    while (true) {
        cout << "\n--- Pinnacle Club Menu ---\n";
        cout << "1. Add President (Div A)\n";
        cout << "2. Add Member (Div A)\n";
        cout << "3. Add Secretary (Div A)\n";
        cout << "4. Display Members (Div A)\n";
        cout << "5. Count Members (Div A)\n";
        cout << "6. Add President (Div B)\n";
        cout << "7. Add Member (Div B)\n";
        cout << "8. Add Secretary (Div B)\n";
        cout << "9. Display Members (Div B)\n";
        cout << "10. Count Members (Div B)\n";
        cout << "11. Concatenate Div A and Div B\n";
        cout << "12. Display Concatenated List\n";
        cout << "13. Exit\n";
        cout << "Enter your choice: ";
        cin >> choice;

        switch (choice) {
        case 1:
            cout << "Enter PRN and Name of President (Div A): ";
            cin >> prn >> name;
            divA.addPresident(prn, name);
            break;
        case 2:
            cout << "Enter PRN and Name of Member (Div A): ";
            cin >> prn >> name;
            divA.addMember(prn, name);
            break;
        case 3:
            cout << "Enter PRN and Name of Secretary (Div A): ";
            cin >> prn >> name;
            divA.addSecretary(prn, name);
            break;
        case 4:
            divA.displayMembers();
            break;
        case 5:
            divA.countMembers();
            break;
        case 6:
            cout << "Enter PRN and Name of President (Div B): ";
            cin >> prn >> name;
            divB.addPresident(prn, name);
            break;
        case 7:
            cout << "Enter PRN and Name of Member (Div B): ";
            cin >> prn >> name;
            divB.addMember(prn, name);
            break;
        case 8:
            cout << "Enter PRN and Name of Secretary (Div B): ";
            cin >> prn >> name;
            divB.addSecretary(prn, name);
            break;
        case 9:
            divB.displayMembers();
            break;
        case 10:
            divB.countMembers();
            break;
        case 11:
            divA.concatenate(divB);
            break;
        case 12:
            divA.displayMembers();
            break;
        case 13:
            cout << "Exiting program...\n";
            return 0;
        default:
            cout << "Invalid choice! Please try again.\n";
        }
    }
}
