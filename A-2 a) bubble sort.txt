#include<iostream>
using namespace std;

void bubblesort(float arr[],int n){
    for(int i=0;i<n-1;i++){
        for(int j=0;j<n-i-1;j++){
            if(arr[j]>arr[j+1]){
                swap(arr[j],arr[j+1]);
            
            }
        }
    }
}

void displayTopFive(float arr[],int n){
    cout<<"\npercentage of top five student: \n"<<endl;
    int count=0;
    for(int i=n-1;i>=0&&count<5;i--){
        cout<<arr[i]<<" ";
        count++;
    
    }
}
int main(){
    int n;
    cout<<"enter the no.of student:"<<endl;
    cin>>n;
    
    
    float percentage[n];
    cout<<"enter the percentage of student:"<<endl;
    for(int i=0;i<n;i++){
        cin>>percentage[i];
        
    }
    bubblesort(percentage ,n);
    cout<<"percentage in ascending order: "<<endl;
    for(int i=0;i<n;i++){
        cout<<percentage[i]<<" ";
        
    }
    cout<<endl;
    
    displayTopFive(percentage,n);
        
    return 0;
}
