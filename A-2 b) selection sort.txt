#include <iostream>
using namespace std ;

void selectionsort(float arr[],int n){
  for(int i=0;i<n-1;i++){
   int minIndex=i;
    for(int j=i+1;j<n;j++){
      if(arr[j]<arr[minIndex]){
        minIndex=j;
      }
    }
    swap(arr[i],arr[minIndex]);
  }
}
void displayTopFive (float arr[],int n){
  cout << "\n percentage of top five students\n" << endl;
  int count =0;
  for(int i=n-1;i>=0&& count<5;i--){
    cout << arr[i] << " ";
    count++;
  }
  cout<<endl;
}
int main(){
  int n;
  cout << "Enter the no.of students :" << endl;
  cin>>n;
  
  float percentage [n];
  cout << "Enter the percentage of students:" << endl;
  for(int i=0;i<n;i++){
   cin>>percentage [i];
  }
  
  selectionsort(percentage,n);
  
  cout << "\n percentage in Ascending order \n " << endl;
  for(int i=0;i<n;i++){
  cout <<percentage [i]<<" ";
  }
  
  displayTopFive(percentage,n);
  
  return 0;

}
