#include <iostream>
using namespace std;

struct Node {
    int data;
    Node* left;
    Node* right;
};

Node* createNode(int value) {
    Node* newNode = new Node();
    newNode->data = value;
    newNode->left = newNode->right = nullptr;
    return newNode;
}

Node* insert(Node* root, int value) {
    if (root == nullptr)
        return createNode(value);

    if (value < root->data)
        root->left = insert(root->left, value);
    else if (value > root->data)
        root->right = insert(root->right, value);

    return root;
}

int height(Node* root) {
    if (root == nullptr)
        return 0;

    int leftHeight = height(root->left);
    int rightHeight = height(root->right);

    return max(leftHeight, rightHeight) + 1;
}

// â¿£ Find minimum value in BST
int findMin(Node* root) {
    if (root == nullptr) {
        cout << "Tree is empty!\n";
        return -1;
    }

    while (root->left != nullptr)
        root = root->left;

    return root->data;
}

void inorder(Node* root) {
    if (root != nullptr) {
        inorder(root->left);
        cout << root->data << " ";
        inorder(root->right);
    }
}

int main() {
    Node* root = nullptr;
    int choice, value, n;

    cout << "Enter number of initial nodes: ";
    cin >> n;

    cout << "Enter " << n << " values:\n";
    for (int i = 0; i < n; i++) {
        cin >> value;
        root = insert(root, value);
    }

    do {
        cout << "\n--- Binary Search Tree Operations ---\n";
        cout << "1. Insert new node\n";
        cout << "2. Find number of nodes in longest path (height)\n";
        cout << "3. Find minimum value in tree\n";
        cout << "4. Display tree (inorder)\n";
        cout << "0. Exit\n";
        cout << "Enter your choice: ";
        cin >> choice;

        switch (choice) {
            case 1:
                cout << "Enter value to insert: ";
                cin >> value;
                root = insert(root, value);
                cout << "Node inserted successfully!\n";
                break;

            case 2:
                cout << "Number of nodes in longest path (height): " << height(root) << endl;
                break;

            case 3:
                cout << "Minimum value in the tree: " << findMin(root) << endl;
                break;

            case 4:
                cout << "Inorder traversal: ";
                inorder(root);
                cout << endl;
                break;

            case 0:
                cout << "Exiting program...\n";
                break;

            default:
                cout << "Invalid choice! Try again.\n";
        }

    } while (choice != 0);

    return 0;
}
