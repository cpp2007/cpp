#include<iostream>
using namespace std;

int Binarysearch(int arr[],int n,int key){
   int start=0;
    int end=n-1;
    while(start<=end){
       int mid=(start+end)/2;
        if(arr[mid]==key){
            return mid;
        }
        else if(arr[mid]<key){
            start=mid+1;
        }
        else{
            end=mid-1;
        }
    }
    return -1;
}
int main(){
    int n;
    cout<<"enter the no.of student:"<<endl;
    cin>>n;
    
    int rollno[n];
    cout<<"enter the rollno.of student (sorted order):"<<endl;
    for(int i=0;i<n;i++){
      cin>>rollno[i];
      
    }
    int key;
    cout<<"enter the search roll no.:"<<endl;
    cin>>key;
    
    int result=Binarysearch(rollno,n,key);
    
    if(result!= -1){
        cout<<"student attend the training program "<<endl;
        
    }
    else{
        cout<<"student does not attend program "<<endl;
    }
        
  return 0;  
    
}
