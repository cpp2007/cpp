#include <iostream>
using namespace std;

struct Node {
    int prn;
    string name;
    Node* next;
};

class PinnacleClub {
private:
    Node* head;

public:
    PinnacleClub() {
        head = nullptr;
    }

    void addPresident(int prn, string name) {
        Node* newNode = new Node{prn, name, head};
        head = newNode;
        cout << "President added successfully.\n";
    }

    void addSecretary(int prn, string name) {
        Node* newNode = new Node{prn, name, nullptr};
        if (head == nullptr) {
            head = newNode;
            return;
        }
        Node* temp = head;
        while (temp->next != nullptr)
            temp = temp->next;
        temp->next = newNode;
        cout << "Secretary added successfully.\n";
    }

    void addMember(int prn, string name) {
        if (head == nullptr || head->next == nullptr) {
            cout << "Please add President and Secretary first.\n";
            return;
        }
        Node* newNode = new Node{prn, name, nullptr};
        Node* temp = head;
        while (temp->next->next != nullptr)
            temp = temp->next;

        newNode->next = temp->next;
        temp->next = newNode;
        cout << "Member added successfully.\n";
    }

    void displayMembers() {
        if (head == nullptr) {
            cout << "No members in the club.\n";
            return;
        }

        cout << "\n--- Pinnacle Club Members ---\n";
        Node* temp = head;
        while (temp != nullptr) {
            cout << temp->prn << " - " << temp->name;
            if (temp == head)
                cout << " (President)";
            else if (temp->next == nullptr)
                cout << " (Secretary)";
            cout << endl;
            temp = temp->next;
        }
    }

    void countMembers() {
        int count = 0;
        Node* temp = head;
        while (temp != nullptr) {
            count++;
            temp = temp->next;
        }
        cout << "Total number of members in the club: " << count << endl;
    }

    void deletePresident() {
        if (head == nullptr) {
            cout << "Club is empty.\n";
            return;
        }

        Node* temp = head;
        head = head->next;
        delete temp;
        cout << "President deleted successfully.\n";
    }

    void deleteSecretary() {
        if (head == nullptr) {
            cout << "Club is empty.\n";
            return;
        }

        if (head->next == nullptr) {
            delete head;
            head = nullptr;
            cout << "Secretary deleted successfully (Club is now empty).\n";
            return;
        }

        Node* temp = head;
        while (temp->next->next != nullptr)
            temp = temp->next;

        delete temp->next;
        temp->next = nullptr;

        cout << "Secretary deleted successfully.\n";
    }

    void deleteMember(int prn) {
        if (head == nullptr) {
            cout << "Club is empty.\n";
            return;
        }

        Node* temp = head;
        Node* prev = nullptr;

        if (temp->prn == prn) {
            cout << "Cannot delete President using this option.\n";
            return;
        }

        while (temp != nullptr && temp->prn != prn) {
            prev = temp;
            temp = temp->next;
        }

        if (temp == nullptr || temp->next == nullptr) {
            cout << "Cannot delete Secretary using this option.\n";
            return;
        }

        prev->next = temp->next;
        delete temp;
        cout << "Member deleted successfully.\n";
    }
};

int main() {
    PinnacleClub club;
    int choice, prn;
    string name;

    while (true) {
        cout << "\n--- Pinnacle Club Menu ---\n";
        cout << "1. Add President\n";
        cout << "2. Add Member\n";
        cout << "3. Add Secretary\n";
        cout << "4. Display Members\n";
        cout << "5. Count Total Members\n";
        cout << "6. Delete President\n";
        cout << "7. Delete Member\n";
        cout << "8. Delete Secretary\n";
        cout << "9. Exit\n";
        cout << "Enter your choice: ";
        cin >> choice;

        switch (choice) {
        case 1:
            cout << "Enter PRN and Name of President: ";
            cin >> prn >> name;
            club.addPresident(prn, name);
            break;
        case 2:
            cout << "Enter PRN and Name of Member: ";
            cin >> prn >> name;
            club.addMember(prn, name);
            break;
        case 3:
            cout << "Enter PRN and Name of Secretary: ";
            cin >> prn >> name;
            club.addSecretary(prn, name);
            break;
        case 4:
            club.displayMembers();
            break;
        case 5:
            club.countMembers();
            break;
        case 6:
            club.deletePresident();
            break;
        case 7:
            cout << "Enter PRN of Member to Delete: ";
            cin >> prn;
            club.deleteMember(prn);
            break;
        case 8:
            club.deleteSecretary();
            break;
        case 9:
            cout << "Exiting program...\n";
            return 0;
        default:
            cout << "Invalid choice! Please try again.\n";
        }
    }
}
